CREATE TABLE `P#COM_USERINFO`(
USERID VARCHAR(32) PRIMARY KEY,
USERNAME VARCHAR(200),
PASSWORD VARCHAR(200),
ISLOCK CHAR(1),
STATUS CHAR(1) default '1',
LOGINTIME TIMESTAMP
);

CREATE OR REPLACE VIEW COM_USERINFO AS
SELECT * FROM `P#COM_USERINFO` WHERE STATUS='1';

CREATE TABLE `P#COM_ROLE`(
ROLEID VARCHAR(32) PRIMARY KEY,
ROLENAME VARCHAR(200),
STATUS CHAR(1) default '1'
);

CREATE OR REPLACE VIEW `COM_ROLE` AS
SELECT * FROM `P#COM_ROLE` WHERE STATUS='1';


CREATE TABLE `P#COM_USERROLE`(

CREATE OR REPLACE VIEW COM_USERROLE AS
