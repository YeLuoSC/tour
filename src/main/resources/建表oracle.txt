CREATE TABLE P#COM_USERINFO(
USERID VARCHAR2(32) PRIMARY KEY,
USERNAME VARCHAR2(200),
PASSWORD VARCHAR2(200),
ISLOCK CHAR(1),
STATUS CHAR(1) default '1',
LOGINTIME TIMESTAMP(6)
);

CREATE OR REPLACE VIEW COM_USERINFO AS
SELECT * FROM P#COM_USERINFO WHERE STATUS='1';

CREATE TABLE P#COM_ROLE(
ROLEID VARCHAR2(32) PRIMARY KEY,
ROLENAME VARCHAR2(200),
STATUS CHAR(1) default '1'
);

CREATE OR REPLACE VIEW COM_ROLE AS
SELECT * FROM P#COM_ROLE WHERE STATUS='1';


CREATE TABLE P#COM_USERROLE(
GUID VARCHAR2(32) PRIMARY KEY,
USERID VARCHAR2(32),
ROLEID VARCHAR2(32),
STATUS CHAR(1) default '1'
);

CREATE OR REPLACE VIEW COM_USERROLE AS
SELECT * FROM P#COM_USERROLE WHERE STATUS='1';

CREATE TABLE P#COM_RESOURCE(
RESOURCEID VARCHAR2(32) PRIMARY KEY,
RESOURCENAME VARCHAR2(200),
URL VARCHAR2(200),
STATUS CHAR(1) default '1'
);

CREATE OR REPLACE VIEW COM_RESOURCE AS
SELECT * FROM P#COM_RESOURCE WHERE STATUS='1';


CREATE TABLE P#COM_ROLERESOURCE(
GUID VARCHAR2(32) PRIMARY KEY,
RESOURCEID VARCHAR2(32),
ROLEID VARCHAR2(32),
STATUS CHAR(1) default '1'
);

CREATE OR REPLACE VIEW COM_ROLERESOURCE AS
SELECT * FROM P#COM_ROLERESOURCE WHERE STATUS='1';
